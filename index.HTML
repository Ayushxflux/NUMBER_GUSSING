OOP DESIGN CONCEPTS

Every pillar of Object-Oriented Programming applied

🧱

ABSTRACTION

The Game class is an abstract base class with pure virtual functions getDifficultyName() and getHint(). It defines what every game mode must do without specifying how.

🔒

ENCAPSULATION

All core data — secretNumber, attempts, score — are private / protected. External access is only through controlled getter methods, preventing data corruption.

🧬

INHERITANCE

EasyGame, MediumGame, and HardGame all inherit from Game using public inheritance. They reuse the core play() loop and only override what's different.

🎭

POLYMORPHISM

A single Game* pointer holds any difficulty object at runtime. Virtual dispatch calls the correct getHint() automatically — no if/else needed in main().

NumberGuessingGame.cpp 
// ── Abstract Base Class ─────────────────────────────────────
class Game {
protected:
    int    secretNumber, attempts, maxAttempts;
    int    minRange, maxRange, score;
    string playerName;

public:
    Game(string name, int minR, int maxR, int maxAtt) { ... }

    // Pure virtual — must be overridden by subclasses
    virtual string getDifficultyName() const = 0;
    virtual string getHint(int guess)   const = 0;

    // Encapsulated getters
    int    getScore()    const { return score; }
    string getPlayerName() const { return playerName; }

    virtual void play() { /* core game loop */ }
};

// ── Derived: Easy (Inheritance + Polymorphism) ──────────────
class EasyGame : public Game {
public:
    EasyGame(string name) : Game(name, 1, 50, 10) {}

    string getDifficultyName() const override {
        return "EASY (1-50, 10 tries)";
    }
    string getHint(int guess) const override {
        int diff = abs(guess - secretNumber);
        if (diff <= 3)  return "🔥 Scorching HOT!";
        if (diff <= 8)  return "♨ WARM!";
        if (diff <= 15) return "❄ Cold.";
        return "🧊 Freezing!";
    }
};

// ── Derived: Medium ─────────────────────────────────────────
class MediumGame : public Game {
public:
    MediumGame(string name) : Game(name, 1, 100, 8) {}
    string getHint(int guess) const override {
        return (guess < secretNumber)
            ? "↑ Too LOW!" : "↓ Too HIGH!";
    }
};

// ── Derived: Hard ───────────────────────────────────────────
class HardGame : public Game {
public:
    HardGame(string name) : Game(name, 1, 200, 6) {}
    string getHint(int guess) const override {
        string dir    = (guess < secretNumber) ? "↑ Higher" : "↓ Lower";
        string parity = (secretNumber % 2 == 0) ? "even" : "odd";
        return dir + ". Answer is " + parity;
    }
};

// ── Scoreboard Class (Encapsulation) ────────────────────────
class Scoreboard {
private:
    string names[10]; int scores[10]; int count;
public:
    Scoreboard() : count(0) {}
    void addEntry(const string& name, int score);
    void display() const;
};

// ── Main: Polymorphism in action ─────────────────────────────
int main() {
    Game* g = nullptr;
    switch (choice) {
        case 1: g = new EasyGame(name);   break;
        case 2: g = new MediumGame(name); break;
        case 3: g = new HardGame(name);   break;
    }
    g->play();   // polymorphic dispatch
    delete g;
}
